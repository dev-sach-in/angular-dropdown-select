<!-- DS-Select JS -->
<script src="dropdown-select.js" defer></script>
<script>
  angular.module('MyApp', ['DropDownSelect'])
    .controller('MainController', ['$scope', function($scope) {
      $scope.items = [
        { id: 1, name: 'Item 1' },
        { id: 2, name: 'Item 2' },
        { id: 3, name: 'Item 3' }
      ];
    }]);
</script>

Single Select

Selected Value: {{selectedValue}}

Code Snippet:
<dropdown-select 
  ds-model="selectedValue" 
  ds-data="items" 
  ds-label="name" 
  ds-value="id" 
  ds-change="onValueChange()"
  ds-class="form-select">
</dropdown-select>
$scope.selectedValue = 2;
$scope.onValueChange = function() {
  console.log('Selected value:', $scope.selectedValue);
};

Multiple Select

Selected Values: {{selectedValues.join(', ')}}

Code Snippet:
<dropdown-select 
  ds-model="selectedValues" 
  ds-data="items" 
  ds-label="name" 
  ds-value="id" 
  ds-multiple="true" 
  ds-change="onValuesChange()"
  ds-class="form-select">
</dropdown-select>
$scope.selectedValues = [1, 3];
$scope.onValuesChange = function() {
  console.log('Selected values:', $scope.selectedValues);
};
